protocol: Pendle V2
domain: "on-chain yield tokenization + interest-rate AMM"
core_idea:
  - Wrap yield-bearing asset into a standardized wrapper (SY)
  - Split SY into principal claim (PT) + yield claim (YT) for a fixed maturity
  - Trade PT vs SY in a single AMM; trade YT via flash-swap routing through the same pool
  - Incentivize LPs via embedded gauges; direct emissions via vePENDLE voting; share fees with voters

contracts_and_modules:
  SY (StandardizedYield, EIP-5115 style):
    purpose: "universal adapter/wrapper for yield-bearing assets"
    responsibilities:
      - wrap/unwap underlying yield asset into SY shares
      - optionally stake/deploy underlying into external protocols to continue earning base yield
      - expose standardized interface for router/market/yield contracts
    properties:
      - permissionless deployment for any yield source, if adapter conforms to SY interface
      - normalizes heterogeneous yield sources (LST/LRT, lending receipts, LP tokens, vault shares, etc.)
  YieldContract (tokenization):
    inputs: "SY amount + chosen expiry"
    outputs: "mint PT + YT pair"
    invariants:
      - collateralization: "PT + YT together represent claim on 1 unit of underlying-at-expiry (via SY)"
      - redemption:
          pre_expiry: "redeem underlying requires PT + YT (re-unite principal + remaining yield claim)"
          post_expiry: "YT ~= 0; redeem underlying requires PT only"
  PT (PrincipalToken):
    semantics: "zero-coupon-like claim to underlying at maturity"
    maturity_behavior:
      - redeem_at_expiry: "1 PT -> 1 unit of underlying (or underlying-denominated claim via SY)"
      - price_path: "typically trades at discount; converges to 1 (par) vs underlying as t -> expiry"
  YT (YieldToken):
    semantics: "claim to all yield generated by underlying between now and expiry"
    behavior:
      - accrues_yield_real_time: "receives yield/rewards stream from the wrapped yield source"
      - value_decay: "tends to 0 at maturity (no remaining future yield)"
  PendleMarket (AMM pool):
    pair: "PT / SY (single pool per maturity)"
    features:
      - custom interest-rate AMM curve with time-decay adjustment
      - embedded geometric-mean TWAP oracle (Uniswap-v3-like observation ring)
      - swap fees auto-compounded into SY side for LPs (fee reinvestment)
      - integrated gauge (no separate staking contract for LP incentives)
    trading_surfaces:
      - direct: "swap PT <-> SY"
      - indirect: "swap YT <-> (SY or underlying) via flash-swap + mint/burn path through same pool"
  Router (PendleRouter):
    role: "stateless public batching/aggregation for UX + gas efficiency"
    interactions:
      - deposits, wrap/unwrap, mint/burn PT/YT, swaps, add/remove liquidity
    security_model: "no special permissions; public utility; integrators can call directly"
  Gauge (embedded in each Market):
    role: "distribute PENDLE emissions + external rewards to LPs"
    stake_model: "LP positions tracked natively in market; no separate staking tx"
    boost_formula:
      activeBalance: "min(LP, 0.4*LP + 0.6*(totalLP * userVe / totalVe))"
      interpretation:
        - "40% unboosted base"
        - "60% boostable via vePENDLE; max boost ~2.5x (per docs narrative)"
  Governance (vePENDLE + controllers):
    vePENDLE:
      lock_range: "1 week to 2 years"
      decay: "linear decay of voting power until unlock"
      utilities:
        - vote emissions allocation across markets (weekly epochs)
        - boost LP rewards (via gauge formula)
        - receive protocol fee share (often only from voted pools, per described model)
    controllers:
      VotingController: "collect votes each epoch"
      GaugeController: "streams PENDLE emissions per-second to market gauges by vote weight"
      VoteController/GaugeController naming: "varies in docs; functional split as above"

market_mechanics:
  tokenization_flow:
    - "Underlying yield asset -> SY wrapper"
    - "SY -> (PT, YT) for expiry"
    - "PT trades vs SY in Market"
    - "YT exposure obtained via flash mechanism routed through Market"
  YT_flash_swap_routing (conceptual):
    identity: "Price(PT) + Price(YT) ~= Price(Underlying) (same expiry, same underlying claim)"
    buy_YT (one canonical path):
      - "flash-borrow SY (or PT) from PT/SY market"
      - "mint PT+YT from borrowed SY"
      - "deliver YT to trader"
      - "sell PT back into market to repay flash loan (+fees)"
    effect: "single PT/SY liquidity base supports both PT and YT trading"
  AMM_design_goals:
    - quote in implied interest rate space (not raw price space)
    - concentrate effective liquidity within expected yield band
    - automatically adapt curve as time-to-expiry shrinks
    - minimize IL for LPs due to strong PT/SY correlation and convergence at expiry
  time_decay_adjustment:
    - "curve re-centers/re-scales over time so PT price converges to par vs underlying at expiry"
    - "capital efficiency increases as expiry approaches (PT ~ underlying)"
  concentrated_liquidity_in_yield_range:
    concept: "like Uniswap v3 concentration but on interest-rate bands"
    property: "LP tokens remain fungible (pool-level concentration, not per-position ranges)"
    benefit: "tighter spreads, lower slippage within expected yield range; adapts if yields move outside"
  fees:
    swap_fees: "earned by LPs; auto-compounded into SY side"
    dynamic_fee: "fee modeled against yield (rate) difference rather than nominal asset delta; lower absolute fee nearer expiry for same rate move"
    protocol_fee (as described): "fee on yield accrued and/or trading; distributed predominantly to vePENDLE voters/lockers (exact params configurable)"

oracles_and_pricing:
  market_TWAP_oracle:
    type: "geometric mean TWAP, observation-based (Uniswap v3 style)"
    usage:
      - "PT/SY average price over configurable window (e.g., 15m/30m/1h)"
      - "integrators should seed observation capacity and wait for window to elapse after market creation"
    properties:
      - "deterministic from market state; no external feed required"
      - "supports manipulation resistance via longer windows"
  PT_to_underlying_pricing:
    composition: "PT/SY * SY/UnderlyingRate"
    note: "SY/Underlying may be 1:1 or use exchange-rate logic depending on yield asset (rebasing, vault share, receipt token)"
  YT_pricing:
    derivation: "YT ~= Underlying - PT (same expiry), assuming consistent unit basis"
    practical: "integrations typically oracle PT; infer YT if needed"
  chainlink_style_adapter (optional):
    purpose: "expose PT price via Chainlink-like interface for protocols expecting Chainlink feeds"
    component: "PendleChainlinkOracleFactory (as referenced)"
